#Outbreak
country_event = {
	id = 999901
	title = "OUTBREAK_START_TITLE"
	desc = "OUTBREAK_START_DESC"
	picture = disease

	fire_only_once = yes
	
	trigger = {
		ai = no
		has_global_flag = zombies_allowed
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
			has_global_flag = first_zombie_invasion
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "OUTBREAK_START_OPT"
		set_global_flag = first_zombie_invasion
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 3
				continent = asia
			}
			country_event = 999902
		}
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 3
				continent = europe
			}
			country_event = 999902
		}
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 3
				OR = {
				continent = north_america
				continent = south_america
				}
			}
			country_event = 999902
		}
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 2
			}
			country_event = 999902
		}
	}
}

#Inital Infection
country_event = {
	id = 999902
	title = "OUTBREAK_START_LOCAL_TITLE"
	desc = "OUTBREAK_START_LOCAL_DESC"
	picture = disease

	trigger = {
		any_neighbor_country = {
			has_country_flag = zombie_outbreak
		}
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
			any_owned_province = {
				has_province_modifier = zombie_outbreak
			}
		}
	}

	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.05
			any_owned = {
			limit = {
				is_colonial = no
			}
			any_neighbor_province = {
			limit = {
			has_province_modifier = zombie_outbreak
			}
		}
	}
	
	option = {
		name = "OUTBREAK_START_LOCAL_OPT"
		set_country_flag = zombie_outbreak
		any_pop = {
			consciousness = 1
			militancy = -2
		}
		random_province = {
			limit = {
				is_capital = no
				is_colonial = no
				is_overseas = no
			}
			any_pop = {
				reduce_pop = 0.95
				consciousness = -3
				militancy = 10
			}
			add_province_modifier = {
				name = zombie_outbreak
				duration = -1
			}
		}
		random_province = {
			limit = {
				is_capital = no
				is_colonial = no
				is_overseas = no
			}

}

#Port Infection
country_event = {
	id = 999903
	title = "OUTBREAK_SPREAD_PORT_TITLE"
	desc = "OUTBREAK_SPREAD_PORT_DESC"
	picture = disease

	trigger = {
		has_global_flag = first_zombie_invasion
		total_num_of_ports = 1
		NOT = {
			border_policy = closed_borders
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = "OUTBREAK_SPREAD_PORT_OPT"
		random_province = {
			limit = {
				port = yes
				has_recent_imigration = 30
				NOT = { has_province_modifier = zombie_outbreak }
			}
			any_pop = {
				consciousness = -3
				militancy = 10
			}
			add_province_modifier = {
				name = zombie_outbreak
				duration = -1
			}
		}
	}
}

province_event = {
	id = 999914
	title = "OUTBREAK_PLAYER_VICTORY_TITLE"
	desc = "OUTBREAK_PLAYER_VICTORY_DESC"

	trigger = {
		has_province_modifier = zombie_outbreak
		NOT = {
			controlled_by = REB
			country_units_in_province = REB
		}
		units_in_province = 10
	}

	mean_time_to_happen = {
		months = 6
		modifier = {
			factor = 1.25
			any_neighbor_province = {
				has_province_modifier = zombie_outbreak
			}
		}
	}

	option = {
		name = "Excellent."
		remove_province_modifier = zombie_outbreak
		any_pop = {
			consciousness = 2
			militancy = -8
		}
	}
}

			}

		months = 3
	}

	option = {
		name = "OUTBREAK_SPREAD_LOCAL_OPT"
			any_owned = {
			limit = {
				has_province_modifier = zombie_outbreak
			}
			any_neighbor_province = {
			limit = {
				NOT = { has_province_modifier = zombie_outbreak }

			}

				consciousness = -3
				militancy = 10

				name = zombie_outbreak
				duration = -1
			}
			}
			}

			limit = {
				has_province_modifier = zombie_outbreak
			}

			limit = {
				neighbor = THIS
			}
			any_neighbor_province = {
			limit = {
				NOT = { has_province_modifier = zombie_outbreak }
			}

			}
			add_province_modifier = {
				name = zombie_outbreak
				duration = -1
			}
			}
			}
			}
		}
}
